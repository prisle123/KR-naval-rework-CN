country_event = {
	id = vnr_ai.0
	title = vnr_ai.0.t
	desc = vnr_ai.0.d

	is_triggered_only = yes

	option = {
		name = vnr_ai.0.0
		set_variable = { vnr_ai_difficulty = 3 }
		hidden_effect = {
			every_country = {
				remove_ideas = {
					vnr_hard
					vnr_normal
				}
			}
		}
		
		every_country = {
			limit = { is_ai = yes }
			add_ideas = vnr_mad
		}
	}
	option = {
		name = vnr_ai.0.1
		set_variable = { vnr_ai_difficulty = 2 }
		hidden_effect = {
			every_country = {
				remove_ideas = {
					vnr_mad
					vnr_normal
				}
			}
		}
		every_country = {
			limit = { is_ai = yes }
			add_ideas = vnr_hard
		}
	}
	option = {
		name = vnr_ai.0.2
		set_variable = { vnr_ai_difficulty = 1 }
		hidden_effect = {
			every_country = {
				remove_ideas = {

				}
			}
		}
		every_country = {
